第一次acm

发表于 2023-09-17 更新于 2023-09-19 阅读次数：

我意识到acm是真正意义上智商的比拼。想要找到能一起讨论题目的人先得看懂题目。只有当你的水平远远高于环境时你的环境才有可能改变。主动改变环境只是徒有其表，内里没有改变，只会不欢而散。
用不欢而散来形容今天很合适，我早就预想到了今天的结果。因为我耐不住寂寞
无法在悄无声息中改变自己。我是一个很虚浮的人
总结一下情况吧。北工大一共有十个队，我们是倒数123的位置（可能真是倒1？）有点丢人。封榜时我们只写出一道题，第一名写出6道题（一共12道）。记录一下情况吧

题目不难，难在理解题意和排序上。人多容易打乱思路，加上刚开始考试时紧张，代码写的有点恶心。我是最开始做这个题的，结果一直做到了离结束还有四十分钟的时候。中途ghj还写了完整的代码但超时了（他完全不会stl）我最后写出了代码但是wa，还是ghj想了个样例给我说新建一个map才搞出来。如果就凭我自己的话大概也只能做出来一道吧

#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
vector<int>v[1000006];
int fa[1000006];
int n, m, a, b;
int x1, x2;
bool flag;
long long eg[1000006];//散列化记边
long long vt[1000006];//散列化记点
long long ans;

int fin(int now) {
    if (fa[now] != now)
        return fa[now] = fin(fa[now]);
    else
        return now;
}

void connect(int A, int B)
{
    int a = fin(A), b = fin(B);
    if (a != b) {
        fa[a] = fa[b] = min(a, b);
    }
}

int main()
{
    //freopen("text_input.txt", "r", stdin);
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
        fa[i] = i;
    for (int i = 0; i < m; i++)
    {
        scanf("%d %d", &a, &b);
        v[a].push_back(b);
        v[b].push_back(a);
        connect(a, b);
    }//建图
    for (int i = 1; i <= n; i++)
    {
        fin(i);
    } 
    for (int i = 1; i <= n; i++)
        eg[fa[i]] += (long long)v[i].size();
    for (int i = 1; i <= n; i++)
        eg[i] /= 2LL;
    for (int i = 1; i <= n; i++)
        vt[fa[i]]++;
    for (int i = 1; i <= n; i++)
        if (vt[i])
            if (eg[i] < (vt[i] * (vt[i] - 1)) / 2)
                ans += ((vt[i] * (vt[i] - 1)) / 2 - eg[i]);
    if (!ans)
    {
        sort(vt + 1, vt + 1 + n);
        for (int i = 1; i <= n; i++)
        {
            if (vt[i] != 0)
            {
                x1 = vt[i];
                x2 = vt[i + 1];
                break;
            }
        }
        ans = x1 * x2;
    }
    cout << ans << endl;
    return 0;
}
