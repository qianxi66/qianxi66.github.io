A:幸运数

题目大意：寻找100000000内的长度为偶数的，前面一半和等于后面一半的数
code： 


#include<iostream>
#include<string>//我忘了tostring在这个头文件里了！
using namespace std;
int num;
int main() {
    int n = 100000000;
    for (int i = 1; i < n; i++)
    {
        string a = to_string(i);
        if (a.size() & 1)continue;
        else {
            int s = a.size() / 2;
            int s1=0, s2 = 0;
            for (int i = 0; i < s; i++)
            {
                s1 += a[i] - '0';
            }
            for (int i = s; i < a.size(); i++)
            {
                s2 += a[i] - '0';
            }
            if (s1 == s2) {
                cout << a << endl;
                num++;
            }
        }
    }
    cout << "答案是"<<num;
}

线段树

发表于 2023-04-03

节点：线段树节点每个代表一个区间，用二分法将每个区间依次减半直到最小，故节点数量是元素数量的二倍。节点内包含三个部分：

struct tree
{
int l, r, mx;//mx为每个区间内要维护的数值（通常为最大，最小值）
}tr[N << 2];

建树：

void build(int u, int l, int r) {//初始化 u:节点编号 从根节点（编号最小的）向下递归
if (l == r)tr[u] = { l,r,0 };
else {
    tr[u] = { l,r,0 };
    int mid = l + r >> 1;
    build(u*2, l, mid), build(u*2+1, mid + 1, r);
    }
}

维护：

inline void pushup(int u) {
tr[u].mx = max(tr[ul].mx, tr[ur].mx);//维护max
}

修改：

void modify(int u, int d, int f) {//当前节点编号 要修改的节点编号 改成什么
if (tr[u].l == d && tr[u].r == d) {
    tr[u].mx = f;
}
else {
    int mid = (tr[u].l + tr[u].r) >> 1;
    if (mid >= d)modify(u*2, d, f);
    if (mid < d)modify(u*2+1, d, f);
    pushup(u);
}
}

查询：

int query(int u, int l, int r) {
if (tr[u].l >= l && tr[u].r <= r) {
    return tr[u].mx;
}
else {
    int mid = tr[u].l + tr[u].r >> 1;
    int t = 0;
    if (mid >= l)t = query(u*2, l, r);
    if (mid < r)t = max(t, query(u*2+1, l, r));
    return t;
}
}
