#include<iostream>
#include<vector>
#include<algorithm>
using namespace std;
vector<int>v[1000006];
int fa[1000006];
int n, m, a, b;
int x1, x2;
bool flag;
long long eg[1000006];//散列化记边
long long vt[1000006];//散列化记点
long long ans;

int fin(int now) {
    if (fa[now] != now)
        return fa[now] = fin(fa[now]);
    else
        return now;
}

void connect(int A, int B)
{
    int a = fin(A), b = fin(B);
    if (a != b) {
        fa[a] = fa[b] = min(a, b);
    }
}

int main()
{
    //freopen("text_input.txt", "r", stdin);
    cin >> n >> m;
    for (int i = 1; i <= n; i++)
        fa[i] = i;
    for (int i = 0; i < m; i++)
    {
        scanf("%d %d", &a, &b);
        v[a].push_back(b);
        v[b].push_back(a);
        connect(a, b);
    }//建图
    for (int i = 1; i <= n; i++)
    {
        fin(i);
    } 
    for (int i = 1; i <= n; i++)
        eg[fa[i]] += (long long)v[i].size();
    for (int i = 1; i <= n; i++)
        eg[i] /= 2LL;
    for (int i = 1; i <= n; i++)
        vt[fa[i]]++;
    for (int i = 1; i <= n; i++)
        if (vt[i])
            if (eg[i] < (vt[i] * (vt[i] - 1)) / 2)
                ans += ((vt[i] * (vt[i] - 1)) / 2 - eg[i]);
    if (!ans)
    {
        sort(vt + 1, vt + 1 + n);
        for (int i = 1; i <= n; i++)
        {
            if (vt[i] != 0)
            {
                x1 = vt[i];
                x2 = vt[i + 1];
                break;
            }
        }
        ans = x1 * x2;
    }
    cout << ans << endl;
    return 0;
}
